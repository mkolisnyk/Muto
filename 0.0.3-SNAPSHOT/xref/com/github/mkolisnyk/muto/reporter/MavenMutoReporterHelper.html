33      public final Map<String, Integer> getOverviewMap(
34              final MutoResult[] resultArray) {
35          Map<String, Integer> result = new HashMap<String, Integer>();
36          int passed = 0;
37          int failed = 0;
38          int errored = 0;
39          for (MutoResult item : resultArray) {
40              if (item.getResults() != null
41                      && item.getResults().size() > 0) {
42                  int errorCount = 0;
43                  for (JUnitTestSuite suite:item.getResults()) {
44                      errorCount += suite.getErrors() + suite.getFailures();
45                  }
46                  if (errorCount > 0) {
47                      errored++;
48                  } else {
49                      passed++;
50                  }
51              } else {
52                  failed++;
53              }
54          }
55          result.put(PASSED, passed);
56          result.put(FAILED, failed);
57          result.put(ERRORED, errored);
58          return result;
59      }

66      @SuppressWarnings("unchecked")
67      public final Map<String, Integer> getTestClassesRating(
68              final MutoResult[] resultArray) {
69          SortedMap<String, Integer> result = new TreeMap<String, Integer>();
70          for (MutoResult item : resultArray) {
71              if (item.getResults() == null
72                      || item.getResults().size() == 0) {
73                  continue;
74              }
75              for (JUnitTestSuite suite:item.getResults()) {
76                  int currentRating = suite.getErrors() + suite.getFailures();
77                  String suiteName = suite.getName();
78                  if (result.containsKey("" + suiteName)) {
79                      currentRating += result.get(suiteName);
80                  }
81                  if (suiteName != null) {
82                      result.put(suiteName, currentRating);
83                  }
84              }
85          }
86          return result;
87      }

94      @SuppressWarnings("unchecked")
95      public final Map<String, Integer> getTestCasesRating(
96              final MutoResult[] resultArray) {
97          SortedMap<String, Integer> result = new TreeMap<String, Integer>();
98          for (MutoResult item : resultArray) {
99              if (item.getResults() == null || item.getResults().size() == 0) {
100                 continue;
101             }
102             for (JUnitTestSuite suite:item.getResults()) {
103                 for (JUnitTestCase test: suite.getTestCases()) {
104                     String testName = suite.getName() + " " + test.getName();
105                     int rating = 0;
106                     if (result.containsKey(testName)) {
107                         rating = result.get(testName);
108                     }
109                     if (test.getError() != null || test.getFailure() != null) {
110                         rating++;
111                     }
112                     result.put(testName, rating);
113                 }
114             }
115         }
116         //return (Map<String, Integer>) entriesSortedByValues(result,false);
117         return result;
118     }
